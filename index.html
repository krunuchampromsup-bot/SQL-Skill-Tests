บทเรียนเรื่อง SQL VIEW

SQL VIEW คือ Virtual Table หรือตารางเสมือนที่สร้างจากคำสั่ง SELECT ช่วยให้การจัดการข้อมูลซับซ้อนกลายเป็นเรื่องง่าย

SELECT: ใช้ดึงข้อมูลจาก VIEW มาแสดงผลเหมือนตารางปกติ
SQL
SELECT * FROM view_name;
INSERT: เพิ่มข้อมูลลงตารางหลักผ่าน VIEW (ต้องเป็น Updatable View)
SQL
INSERT INTO view_name (col1, col2) VALUES (val1, val2);
UPDATE: แก้ไขข้อมูลในตารางหลักผ่านชื่อ VIEW
SQL
UPDATE view_name SET col1 = val1 WHERE condition;
DELETE: ลบแถวข้อมูลออกจากตารางหลักผ่าน VIEW
SQL
DELETE FROM view_name WHERE condition;

โจทย์ที่ 1: การสร้าง VIEW

จงสร้าง VIEW ชื่อ it_staff จากตาราง employees เฉพาะคนที่อยู่แผนก 'IT'

CREATE VIEW it_staff AS SELECT * FROM employees WHERE department = 'IT';

โจทย์ที่ 2: การดึงข้อมูลจาก VIEW

จงเขียนคำสั่งแสดงข้อมูลทั้งหมดจาก VIEW ที่ชื่อว่า it_staff

SELECT * FROM it_staff;

โจทย์ที่ 3: การแก้ไขข้อมูลผ่าน VIEW

จงอัปเดตเงินเดือน (salary) ใน VIEW it_staff ให้เป็น 50000 สำหรับพนักงานที่มี id = 10

UPDATE it_staff SET salary = 50000 WHERE id = 10;
